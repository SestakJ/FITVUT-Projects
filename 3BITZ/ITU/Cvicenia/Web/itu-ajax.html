<script type="text/javascript">

/***
* XMLHttpRequest object constructor (for compatibility with various browsers)
*/

function createXmlHttpRequestObject() {
var xmlhttp;
try {
xmlHttp = new XMLHttpRequest(); //should work on all browsers except IE6 or older
} catch (e) {
try {
xmlHttp = new ActiveXObject("Microsoft.XMLHttp"); //browser is IE6 or older
} catch (e) {
// ignore error
}
}
if (!xmlHttp) {
alert ("Error creating the XMLHttpRequest object.");
} else {
return xmlHttp;
}
}

function uploadData() {
try {
var xmlHttp = createXmlHttpRequestObject();//stores XMLHttpRequestObject
var msg = "data=" + document.getElementById("newMessageString").value;
xmlHttp.open("POST", "http://www.stud.fit.vutbr.cz/~xmlich02/itu-ajax/api.php", true);
xmlHttp.setRequestHeader("Content-Type", "application/x-www-form-urlencoded;");
xmlHttp.setRequestHeader("Content-lenght", msg.lenght);
xmlHttp.setRequestHeader("Connection", "close");
xmlHttp.onreadystatechange = function() {
if ((xmlHttp.readyState == 4) && (xmlHttp.status = 200)) {
downloadData();
}
}
xmlHttp.send(msg);
document.getElementById("newMessageString").value = ""

} catch (e) {
alert(e.toString());
}

return false; // to avoid default form submit behavior
}

var last_msg_id = 0;

function downloadData() {
var xmlHttp = createXmlHttpRequestObject();
xmlHttp.open("POST", "http://www.stud.fit.vutbr.cz/~xmlich02/itu-ajax/api.php", true );
xmlHttp.setRequestHeader("Content-Type", "application/x-www-form-urlencoded;");
xmlHttp.setRequestHeader("Connection", "close");
xmlHttp.onreadystatechange = function() {
if ((xmlHttp.readyState == 4) && (xmlHttp.status = 200))
{
var json_data = JSON.parse(xmlHttp.responseText);
for (var i in json_data) {
if (json_data[i].id <= last_msg_id) continue;

var span_node = document.createElement("span");
span_node.innerHTML = "<b class ='" + (json_data[i].login == "xbolva00" ? "me" : "") + "'>" + json_data[i].login + "</b>: " + json_data[i].cnt;
document.getElementById("chatArea").appendChild(span_node);
document.getElementById("chatArea").scrollTop = document.getElementById("chatArea").scrollHeight;
document.getElementById("status").innerHTML = json_data[i].id;
last_msg_id = json_data[i].id;
}
}
}
xmlHttp.send(null);

}

setInterval("downloadData()", 3000);

downloadData();

</script>
